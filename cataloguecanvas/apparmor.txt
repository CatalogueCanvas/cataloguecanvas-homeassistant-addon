#include <tunables/global>

# AppArmor profile for the CatalogueCanvas add-on (root under s6-overlay -> run.sh -> uvicorn).

profile cataloguecanvas flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>
  #include <abstractions/nameservice>
  #include <abstractions/openssl>
  #include <abstractions/python>

  # Networking, signals, and capabilities the add-on needs to run as root under s6.
  network inet tcp,
  network inet6 tcp,
  signal (receive) set=(kill,term,int,hup,cont),
  capability chown,
  capability dac_override,
  capability fowner,
  capability setgid,
  capability setuid,

  # Interpreters, system binaries, and the s6-overlay supervision tree.
  /bin/** ix,
  /usr/bin/** ix,
  /usr/local/bin/** ix,
  /usr/local/sbin/** ix,
  /sbin/** ix,
  /init ix,
  /package/** ix,
  /command/** ix,
  /etc/s6-overlay/** rix,
  /run/{s6,s6-rc,service}/** rwix,

  # Application tree, native libraries, and fonts/certs.
  /app/** rwix,
  /usr/lib/** mr,
  /usr/local/lib/** mr,
  /usr/share/** r,
  /etc/** r,

  # Persistent and mapped Home Assistant volumes.
  /config/** rw,
  /data/** rw,
  /share/** rw,
  /media/** rw,
  /homeassistant/** r,

  # Working/temp/proc/dev that the interpreter and app need.
  /tmp/** rwk,
  /run/{,**} rwk,
  /proc/*/{fd,status,stat,mounts,cmdline} r,
  /proc/sys/kernel/random/uuid r,
  /dev/tty rw,
  /dev/urandom r,
  /dev/null rw,

  deny /proc/kcore rwklx,
  deny /proc/sysrq-trigger rwklx,
  deny /sys/firmware/** rwklx,
}
